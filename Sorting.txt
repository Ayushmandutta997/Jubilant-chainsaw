int partition(int l,int h)
{
  int i=l;
  int j=h;
  int pi=a[l];
  while(i<j)
  { 
   do
   {
     i++;
   }while(pi>=a[i]);
   do
   {
     j--;
   }while(pi<a[j]);
   if(i<j)
   swap(a[i],a[j]);
  }
  swap(a[l],a[j]);
  return j;
}
void quicksort(int l,int h)
{
  if(l<h)
  {
    int pi=partition(l,h);
    quicksort(l,pi);
    quicksort(pi+1,h);
  }
}
